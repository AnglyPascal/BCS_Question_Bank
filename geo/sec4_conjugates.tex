	\theo{https://artofproblemsolving.com/community/c2771h1181729_isogonal_line_lemma_updated}{Isogonal Line Lemma}{Let $AP,AQ$ are isogonal lines with respect to $\angle BAC$. Let $BP \cap CQ = F$ and $BQ \cap CP = E$. Then $AE,AF$ are isogonal lines with respect to $\angle BAC$.
		\figdf{.55}{Isogonal_Lemma}{}}
	
	\proof{\[A(B, F; P, X) = (B, F; P, X) = C(B, Q; E, X) = (B, Q; E, X) = (X, E; Q, B)\]
	So if we define a projective transformation that swaps isogonal lines wrt $ \angle BAC $, we see $ AE, AF $ are conjugates of each other.}

Let the perpendicular bisectors of $ {AB}$ and $ {AC}$ intersect ray $ AM$ in points $ D$ and $ E$ respectively, and let lines $ BD$ and $ CE$ intersect in point $ F$, inside of triangle $ ABC$. Prove that points $ A$, $ N$, $ F$, and $ P$ all lie on one circle.
	\figdf{.5}{USAMO_2008_P2}{USAMO 2008 P2}}
	
	
		\begin{solution}[Phantom Point]
			First assume $ F\in BD, \text{ and } F=T' $ (Where $ T' $ comes from \autoref{lemma:Most Important Symmedian Property}, and prove that $ F\in CE $.)
		\end{solution}
		\begin{solution}[Isogonal Conjugate]
			Construct the isogonal conjugate of $ F $, which is the intersection of the circles touching $ BC $ and passing through $ A, B $ and $ A, C $.
		\end{solution}
		\solu{Using \autoref{theorem:Isogonal Line Lemma} by taking the reflections of $ B , C $ over $ D, F $}
